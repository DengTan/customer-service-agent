interface QuickReply {
  id: string;
  title: string;
  content: string;
  category?: string;
  scope: string;
  variables?: unknown[];
  usage_count: number;
  creator_id?: string;
  created_at: string;
  updated_at?: string;
}

interface QuickRepliesPanelProps {
  /** 闈㈡澘鎵撳紑鐘舵€侊紝鐢卞閮?Dialog 鎺у埗 */
  open?: boolean;
  /** 闈㈡澘鐘舵€佸彉鏇村洖璋?*/
  onOpenChange?: (open: boolean) => void;
  /** 閫変腑璇濇湳鍥炶皟 */
  onSelect?: (reply: QuickReply) => void;
  /** 鏄惁鏄剧ず瀵煎叆瀵煎嚭鎿嶄綔 */
  showActions?: boolean;
  className?: string;
}

export function QuickRepliesPanel({
  open = true,
  onOpenChange,
  onSelect,
  showActions = true,
  className,
}: QuickRepliesPanelProps) {
  const [replies, setReplies] = useState<QuickReply[]>([]);
  const [loading, setLoading] = useState(false);
  const [search, setSearch] = useState("");
  const [categoryFilter, setCategoryFilter] = useState("all");
  const [scopeFilter, setScopeFilter] = useState("all");

  const [editingId, setEditingId] = useState<string | null>(null);
  const [deletingId, setDeletingId] = useState<string | null>(null);
  const [saving, setSaving] = useState(false);
  const [importDialogOpen, setImportDialogOpen] = useState(false);
  const [importing, setImporting] = useState(false);
  const [copiedId, setCopiedId] = useState<string | null>(null);
  const [confirmDeleteId, setConfirmDeleteId] = useState<string | null>(null);
  const [form, setForm] = useState({
    title: "",
    content: "",
    category: "鍏朵粬",
    scope: "global",

  // 鍔犺浇璇濇湳鍒楄〃 (API绔瓫閫?
  const loadReplies = useCallback(async () => {
    setLoading(true);
    try {
      const params = new URLSearchParams();
      if (categoryFilter && categoryFilter !== "all" && categoryFilter !== "鍏ㄩ儴") {
        params.set("category", categoryFilter);
      }
      if (scopeFilter && scopeFilter !== "all" && scopeFilter !== "鍏ㄩ儴") {
        params.set("scope", scopeFilter);

      if (search.trim()) params.set("search", search.trim());
      const url = `${API_BASE}${params.toString() ? `?${params}` : ""}`;
      const res = await fetch(url);
      if (res.ok) {
        const data = await res.json();
        setReplies(data.items || []);
      }
    } catch (error) {
      logger.error('鍔犺浇璇濇湳澶辫触', { error });
    } finally {
      setLoading(false);
    }
  }, [categoryFilter, scopeFilter, search]);

    loadReplies();

    const cats = new Set(replies.map((r) => r.category).filter((c): c is string => Boolean(c)));
    return Array.from(cats).sort();

    setForm({ title: "", content: "", category: "鍏朵粬", scope: "global" });
    setDialogOpen(true);
  };

  // 鎵撳紑缂栬緫
  const openEdit = (reply: QuickReply) => {
    setEditingId(reply.id);
    setForm({
      title: reply.title,
      content: reply.content,
      category: reply.category || "鍏朵粬",
      scope: reply.scope,
    });
    setDialogOpen(true);
  };

  // 淇濆瓨
  const handleSave = async () => {
    if (!form.title.trim() || !form.content.trim()) {
      toast.error("璇峰～鍐欐爣棰樺拰鍐呭");
      return;
    }

    setSaving(true);
    try {
      const url = editingId ? `${API_BASE}?id=${editingId}` : API_BASE;
      const method = editingId ? "PUT" : "POST";
      const res = await fetch(url, {
        method,
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(editingId ? { ...form, id: editingId } : form),
      });

      if (res.ok) {
        toast.success(editingId ? "璇濇湳宸叉洿鏂? : "璇濇湳宸插垱寤?);
        setDialogOpen(false);
        if (editingId) {
          // 缂栬緫锛氬埛鏂板綋鍓嶉〉
          await reset();
        } else {
          // 鏂板缓锛氬鍔犳€绘暟骞跺埛鏂?          setTotal((n) => n + 1);
          await reset();
        }
      } else {
        toast.error("淇濆瓨澶辫触");
      }
    } catch (error) {
      toast.error("淇濆瓨澶辫触");
    } finally {
      setSaving(false);
    }

  // 鍒犻櫎

        updateItems((prev) => prev.filter((r) => r.id !== id));

      }

      toast.error("鍒犻櫎澶辫触");
    } finally {
      setDeletingId(null);
    }

    e.stopPropagation();

    }
  };

  // 閫変腑璇濇湳
  const handleSelect = (reply: QuickReply) => {
    if (onSelect) {
      onSelect(reply);
    }

  const handleExport = async (format: "xlsx" | "csv") => {
    try {
      const res = await fetch(`${API_BASE}/export?format=${format}`);

      const blob = await res.blob();
      const url = URL.createObjectURL(blob);
      const a = document.createElement("a");
      a.href = url;
      a.download = `璇濇湳搴揰${new Date().toLocaleDateString("zh-CN").replace(/\//g, "-")}.${format}`;
      a.click();
      URL.revokeObjectURL(url);
      toast.success(`宸插鍑轰负 ${format.toUpperCase()}`);

      toast.error("瀵煎嚭澶辫触");

        body: formData,

      if (res.ok) {

        } else {
          toast.success(successMsg, { id: loadingToast });
        }
        setImportDialogOpen(false);
        await reset();
      } else {
        toast.error(data.error || "瀵煎叆澶辫触", { id: loadingToast });
      }
    } catch (error) {
      toast.error("瀵煎叆澶辫触");
    } finally {
      setImporting(false);
      // 閲嶇疆 file input
      setTimeout(() => {
        document.getElementById("import-file")?.setAttribute("value", "");
      }, 100);
    }

  return (
    <div className={cn("flex flex-col h-full bg-background", className)}>
      {/* 鎼滅储绛涢€夋爮 */}

        <div className="relative flex-1">
          <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-muted-foreground" />
          <Input
            placeholder="鎼滅储璇濇湳..."
            value={search}
            onChange={(e) => setSearch(e.target.value)}
            className="pl-9 h-8 text-sm bg-background"
          />
        </div>

        <Select value={categoryFilter} onValueChange={setCategoryFilter}>
          <SelectTrigger className="w-[100px] h-8 text-xs">
            <SelectValue placeholder="鍒嗙被" />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="all">鍏ㄩ儴鍒嗙被</SelectItem>
            {categories.map((cat) => (
              <SelectItem key={cat} value={cat}>
                {cat}
              </SelectItem>
            ))}
          </SelectContent>
        </Select>

        <Select value={scopeFilter} onValueChange={setScopeFilter}>
          <SelectTrigger className="w-[80px] h-8 text-xs">
            <SelectValue />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="all">鍏ㄩ儴</SelectItem>
            <SelectItem value="global">鍏ㄥ眬</SelectItem>
            <SelectItem value="agent">鍧愬腑</SelectItem>
            <SelectItem value="ai">AI</SelectItem>
          </SelectContent>
        </Select>

        <Button size="sm" onClick={openCreate} className="h-8 gap-1 shrink-0">
          <Plus className="w-3.5 h-3.5" />
          鏂板缓
        </Button>
      </div>

      {/* 缁熻鍜屾搷浣滄爮 */}
      {showActions && (
        <div className="flex items-center justify-between px-3 py-2 border-b bg-muted/20">
          <div className="flex items-center gap-3 text-xs text-muted-foreground">
            <span>
              鍏?<span className="font-medium text-foreground">{total}</span> 鏉¤瘽鏈?              {replies.length < total && (
                <span className="text-muted-foreground/60">锛堝凡鍔犺浇 {replies.length}锛?/span>
              )}
            </span>
            {(search || categoryFilter !== "all" || scopeFilter !== "all") && (
              <span
                className={loading ? "text-orange-600" : "text-green-600"}
                aria-live="polite"
              >
                {loading ? "绛涢€変腑" : "宸茬瓫閫?}
              </span>
            )}
          </div>

          <div className="flex items-center gap-1">
            <DropdownMenu>
              <DropdownMenuTrigger asChild>
                <Button variant="ghost" size="sm" className="h-7 text-xs gap-1">
                  <Download className="w-3 h-3" />
                  瀵煎嚭

              <DropdownMenuContent align="end">
                <DropdownMenuItem onClick={() => handleExport("xlsx")}>
                  <FileSpreadsheet className="w-3.5 h-3.5 mr-2" />
                  Excel 鏍煎紡
                </DropdownMenuItem>
                <DropdownMenuItem onClick={() => handleExport("csv")}>
                  <FileText className="w-3.5 h-3.5 mr-2" />
                  CSV 鏍煎紡
                </DropdownMenuItem>
              </DropdownMenuContent>
            </DropdownMenu>

            <Button
              variant="ghost"
              size="sm"
              className="h-7 text-xs gap-1"
              onClick={() => setImportDialogOpen(true)}
            >
              <Upload className="w-3 h-3" />
              瀵煎叆
            </Button>
          </div>
        </div>
      )}

      <ReplyList
        replies={replies}
        isInitialLoading={isInitialLoading}
        isLoadingMore={isLoadingMore}
        hasMore={hasMore}
        onLoadMore={loadMore}
        onSelect={onSelect ? handleSelect : undefined}
        onCopy={handleCopy}
        onEdit={openEdit}
        onDelete={(id) => setConfirmDeleteId(id)}
        onCreateFirst={openCreate}
        copiedId={copiedId}
        deletingId={deletingId}
        search={search}
        categoryFilter={categoryFilter}
        scopeFilter={scopeFilter}
      />

      {/* 鏂板缓/缂栬緫瀵硅瘽妗?*/}
      <Dialog open={dialogOpen} onOpenChange={setDialogOpen}>
        <DialogContent className="sm:max-w-md">
          <DialogHeader>
            <DialogTitle>{editingId ? "缂栬緫璇濇湳" : "鏂板缓璇濇湳"}</DialogTitle>
            <DialogDescription className="text-xs">
              {editingId ? "淇敼璇濇湳鍐呭" : "鍒涘缓鏂扮殑蹇嵎鍥炲璇濇湳"}
            </DialogDescription>
          </DialogHeader>

          <div className="space-y-4 py-2">
            <div className="space-y-1.5">
              <label className="text-xs font-medium text-foreground">鏍囬</label>
              <Input
                placeholder="杈撳叆璇濇湳鏍囬"
                value={form.title}
                onChange={(e) => setForm({ ...form, title: e.target.value })}
              />
            </div>

            <div className="space-y-1.5">
              <label className="text-xs font-medium text-foreground">鍐呭</label>
              <textarea
                className="w-full min-h-[120px] px-3 py-2 rounded-md border border-input bg-background text-sm ring-offset-background placeholder:text-muted-foreground focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring resize-none"

                    <SelectValue />

                  <SelectContent>

                    <SelectItem value="global">

            {/* 閲嶅澶勭悊璇存槑 */}
            <div className="flex items-start gap-2 text-xs text-muted-foreground bg-amber-50 rounded-lg p-2">
              <Info className="w-3.5 h-3.5 text-amber-600 shrink-0 mt-0.5" />
              <span>鐩稿悓鏍囬鐨勮瘽鏈細琚嚜鍔ㄨ烦杩囷紝涓嶄細閲嶅鍒涘缓</span>
            </div>
          </div>

          <DialogFooter>
            <Button variant="outline" onClick={() => setImportDialogOpen(false)}>
              鍏抽棴
            </Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>

      {/* 鍒犻櫎纭寮圭獥 */}

        <AlertDialogContent>

            <AlertDialogDescription>
              纭畾瑕佸垹闄よ繖鏉¤瘽鏈悧锛熸鎿嶄綔鏃犳硶鎾ら攢銆?            </AlertDialogDescription>
          </AlertDialogHeader>
          <AlertDialogFooter>
            <AlertDialogCancel onClick={() => setConfirmDeleteId(null)}>
              鍙栨秷
            </AlertDialogCancel>
            <AlertDialogAction
              onClick={() => {
                if (confirmDeleteId) {
                  handleDelete(confirmDeleteId);
                  setConfirmDeleteId(null);
                }
              }}
              className="bg-destructive text-destructive-foreground hover:bg-destructive/90"
            >
              鍒犻櫎
            </AlertDialogAction>
          </AlertDialogFooter>
        </AlertDialogContent>
      </AlertDialog>
    </div>
